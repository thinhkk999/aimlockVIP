function getTimeRemaining(expiresAt) {
  if (!expiresAt) return { text: "Vĩnh viễn", expired: false, days: -1, hours: -1, minutes: -1 };
  const now = new Date();
  const expireDate = new Date(expiresAt);
  if (expireDate <= now) return { text: "Đã hết hạn", expired: true, days: 0, hours: 0, minutes: 0 };
  const diffMs = expireDate - now;
  const days = Math.floor(diffMs / (1000 * 60 * 60 * 24));
  const hours = Math.floor((diffMs % (86400000)) / (3600000));
  const minutes = Math.floor((diffMs % 3600000) / (60000));
  if (days > 0) return { text: `${days} ngày ${hours} giờ`, expired: false, days, hours, minutes };
  if (hours > 0) return { text: `${hours} giờ ${minutes} phút`, expired: false, days, hours, minutes };
  return { text: `${minutes} phút`, expired: false, days, hours, minutes };
}

async function initAudio() {
  if (audioCtx) return;
  audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  function createBeepSound(freq, duration) {
    const sampleRate = audioCtx.sampleRate;
    const samples = duration * sampleRate;
    const buffer = audioCtx.createBuffer(1, samples, sampleRate);
    const data = buffer.getChannelData(0);
    for (let i = 0; i < samples; i++) {
      const t = i / sampleRate;
      data[i] = Math.sin(2 * Math.PI * freq * t) * Math.exp(-3 * t);
    }
    return buffer;
  }
  clickSoundBuffer = createBeepSound(880, 0.15);
  toggleSoundBuffer = createBeepSound(440, 0.2);
}

function playSound(type = 'click') {
  if (!audioCtx) return;
  if (audioCtx.state === 'suspended') audioCtx.resume();
  const buffer = type === 'click' ? clickSoundBuffer : toggleSoundBuffer;
  if (!buffer) return;
  const source = audioCtx.createBufferSource();
  source.buffer = buffer;
  source.connect(audioCtx.destination);
  source.start();
}

function getHWID() {
  const navigatorInfo = `${navigator.userAgent}|${navigator.language}|${navigator.platform}|${screen.width}x${screen.height}|${screen.colorDepth}|${navigator.hardwareConcurrency || 4}`;
  let hash = 0;
  for (let i = 0; i < navigatorInfo.length; i++) {
    hash = ((hash << 5) - hash) + navigatorInfo.charCodeAt(i);
    hash = hash & hash;
  }
  return Math.abs(hash).toString(16).toUpperCase();
}

async function fetchFirebase(path, method = 'GET', data = null) {
  const url = `${FIREBASE_URL}${path}.json`;
  const options = { method: method, headers: { 'Content-Type': 'application/json' } };
  if (data) options.body = JSON.stringify(data);
  try {
    const response = await fetch(url, options);
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return await response.json();
  } catch (error) {
    console.error('Firebase Error:', error);
    return null;
  }
}

function drawChart(ctx, canvas, data, color) {
  if (!ctx || !canvas) return;
  const w = canvas.width;
  const h = canvas.height;
  ctx.clearRect(0, 0, w, h);
  if (data.length < 2) return;
  const step = w / (data.length - 1);
  ctx.beginPath();
  ctx.moveTo(0, h - (data[0] / 100) * h);
  for (let i = 1; i < data.length; i++) ctx.lineTo(i * step, h - (data[i] / 100) * h);
  ctx.strokeStyle = color; ctx.lineWidth = 2; ctx.stroke();
  ctx.lineTo(w, h); ctx.lineTo(0, h); ctx.fillStyle = color + '20'; ctx.fill();
}

function isAndroid() {
  return /Android/i.test(navigator.userAgent || "");
}

function isIOS() {
  return /iPhone|iPad|iPod/i.test(navigator.userAgent || "");
}

function tryOpenUrl(url) {
  try {
    const iframe = document.createElement("iframe");
    iframe.style.display = "none";
    iframe.src = url;
    document.body.appendChild(iframe);
    setTimeout(function () {
      iframe.remove();
    }, 1200);
  } catch (e) {}
  try {
    window.location.href = url;
  } catch (e) {}
}

function launchGameDirect(target) {
  const isMax = target === "ffmax";
  const scheme = isMax ? "freefiremax://" : "freefire://";
  const androidPackage = isMax ? "com.dts.freefiremax" : "com.dts.freefireth";

  if (isAndroid()) {
    const intentUrl = `intent://${androidPackage}#Intent;scheme=${isMax ? "freefiremax" : "freefire"};package=${androidPackage};end`;
    tryOpenUrl(intentUrl);
  } 
  else if (isIOS()) {
    tryOpenUrl(scheme);
  }
}

function initParticles() {
  const pCanvas = document.getElementById('particles-canvas');
  if (!pCanvas) return;
  const pCtx = pCanvas.getContext('2d');
  let pWidth = window.innerWidth, pHeight = window.innerHeight;
  pCanvas.width = pWidth; pCanvas.height = pHeight;
  let particles = [];
  for (let i = 0; i < 60; i++) {
    particles.push({
      x: Math.random() * pWidth, y: Math.random() * pHeight,
      vx: (Math.random() - 0.5) * 0.3, vy: (Math.random() - 0.5) * 0.2,
      radius: Math.random() * 2 + 1, alpha: Math.random() * 0.4 + 0.1
    });
  }
  function animate() {
    if (!pCtx) return;
    pCtx.clearRect(0, 0, pWidth, pHeight);
    particles.forEach(p => {
      p.x += p.vx; p.y += p.vy;
      if (p.x < 0) p.x = pWidth; if (p.x > pWidth) p.x = 0;
      if (p.y < 0) p.y = pHeight; if (p.y > pHeight) p.y = 0;
      pCtx.beginPath(); pCtx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
      pCtx.fillStyle = `rgba(255,42,42,${p.alpha})`;
      pCtx.fill();
    });
    requestAnimationFrame(animate);
  }
  animate();
  window.addEventListener('resize', () => { pWidth = window.innerWidth; pHeight = window.innerHeight; pCanvas.width = pWidth; pCanvas.height = pHeight; });
}

function initCrosshair() {
  const xhToggle = document.getElementById('crosshair-toggle');
  const xhOverlay = document.getElementById('virtual-crosshair-overlay');
  const xhSizeSlider = document.getElementById('xh-size-slider');
  const xhSizeVal = document.getElementById('xh-size-val');
  const xhColorPicker = document.getElementById('xh-color-picker');
  const xhDraw = document.getElementById('xh-draw');
  const xhXSlider = document.getElementById('xh-x-slider');
  const xhYSlider = document.getElementById('xh-y-slider');
  const xhXVal = document.getElementById('xh-x-val');
  const xhYVal = document.getElementById('xh-y-val');
  
  function update() {
    const size = xhSizeSlider ? parseFloat(xhSizeSlider.value) : 1;
    const color = xhColorPicker ? xhColorPicker.value : '#ff0000';
    const posX = xhXSlider ? parseInt(xhXSlider.value) : 0;
    const posY = xhYSlider ? parseInt(xhYSlider.value) : 0;
    if(xhSizeVal) xhSizeVal.innerText = size + "x";
    if(xhXVal) xhXVal.innerText = posX;
    if(xhYVal) xhYVal.innerText = posY;
    if(xhDraw && xhOverlay) {
      xhOverlay.style.transform = `translate(calc(-50% + ${posX}px), calc(-50% + ${posY}px))`;
      xhDraw.style.transform = `scale(${size})`;
      xhDraw.style.borderColor = color;
      const dot = xhDraw.children[0];
      if(dot) dot.style.backgroundColor = color;
    }
  }
  
  if(xhToggle) xhToggle.addEventListener('change', () => {
    if(xhOverlay) xhOverlay.style.display = xhToggle.checked ? 'block' : 'none';
    if(xhToggle.checked) { playSound('toggle'); runUniversalModal("KÍCH HOẠT TÂM ẢO", "fa-bullseye"); }
  });
  if(xhSizeSlider) xhSizeSlider.addEventListener('input', update);
  if(xhColorPicker) xhColorPicker.addEventListener('input', update);
  if(xhXSlider) xhXSlider.addEventListener('input', update);
  if(xhYSlider) xhYSlider.addEventListener('input', update);
  update();
}

function initThemePicker() {
  const savedTheme = localStorage.getItem('app_theme') || 'red';
  document.body.classList.add(`theme-${savedTheme}`);
  document.querySelectorAll('.theme-btn').forEach(btn => {
    const theme = btn.dataset.theme;
    if (theme === savedTheme) btn.classList.add('active-theme');
    btn.addEventListener('click', () => {
      const newTheme = btn.dataset.theme;
      document.body.classList.remove('theme-red', 'theme-cyan', 'theme-purple', 'theme-dark');
      document.body.classList.add(`theme-${newTheme}`);
      localStorage.setItem('app_theme', newTheme);
      document.querySelectorAll('.theme-btn').forEach(b => b.classList.remove('active-theme'));
      btn.classList.add('active-theme');
      showToast(`Đã chuyển sang giao diện ${newTheme.toUpperCase()}`, 'info');
      playSound('click');
    });
  });
}
